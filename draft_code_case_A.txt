model = cp_model.CpModel()

# Variabili booleane x[w, d, s]
x = {}
for w in WORKER_IDS:
    for d in range(NUM_DAYS):
        for s in SHIFT_CODES:
            x[(w, d, s)] = model.NewBoolVar(f'x_{w}_{d}_{s}')

# Vincoli HARD

# 1. Max 36 ore settimanali e Max 6 giorni lavorati a settimana
for w in WORKER_IDS:
    for t in range(NUM_DAYS - 6):  # finestre di 7 giorni: t, t+1, ..., t+6
        finestra = range(t, t + 7)
        model.Add(sum(SHIFT_HOURS[s] * x[(w, d, s)]
                      for d in finestra for s in SHIFT_CODES) <= 36)
        model.Add(sum(x[(w, d, s)]
                      for d in finestra for s in SHIFT_CODES) <= 6)

# 2. Esattamente 25 turni/mese per lavoratore (conteggiati col peso)
for w in WORKER_IDS:
    model.Add(sum(SHIFT_WEIGHT[s] * x[(w, d, s)]
                  for d in range(NUM_DAYS) for s in SHIFT_CODES) == 25)

# 3. Max 1 turno al giorno per lavoratore
for w in WORKER_IDS:
    for d in range(NUM_DAYS):
        model.AddAtMostOne(x[(w, d, s)] for s in SHIFT_CODES)

# 3b. Divieto Notte(d) -> Mattina(d+1)
for w in WORKER_IDS:
    for d in range(NUM_DAYS - 1):
        model.Add(x[(w, d, 'N')] + x[(w, d + 1, 'M')] <= 1)

# 5. Dopo OGNI turno di Notte: 2 giorni di riposo TOTALE
for w in WORKER_IDS:
    for d in range(NUM_DAYS):
        for k in range(1, 3):  # k=1 e k=2 (i due giorni di riposo)
            if d + k < NUM_DAYS:
                for s in SHIFT_CODES:
                    model.Add(x[(w, d, 'N')] + x[(w, d + k, s)] <= 1)

# Turni vietati (FORBIDDEN_SHIFTS)
for w in WORKER_IDS:
    for s in FORBIDDEN_SHIFTS[w]:
        for d in range(NUM_DAYS):
            model.Add(x[(w, d, s)] == 0)

# Caso A: almeno 2 lavoratori assegnati a OGNI turno di OGNI giorno
for d in range(NUM_DAYS):
    for s in SHIFT_CODES:
        model.Add(sum(x[(w, d, s)] for w in WORKER_IDS) >= 2)

# Funzione Obiettivo
obiettivo = []
for w in WORKER_IDS:
    for d in range(NUM_DAYS):
        for s in SHIFT_CODES:
            peso = int(round(PREFERENCES[w]['satisfaction_weights'][s] * 10))
            if d in UNDESIRED_DAYS[w]:
                peso -= int(round(UNDESIRED_DAY_PENALTY * 10))
            obiettivo.append(peso * x[(w, d, s)])
model.Maximize(sum(obiettivo))

# Risoluzione
solver = cp_model.CpSolver()
solver.parameters.max_time_in_seconds = MAX_TIME
solver.parameters.num_search_workers = 8
solver.parameters.log_search_progress = False

status = solver.Solve(model)

# Popolamento di RESULT_SCHEDULE e SOLVER_STATUS
RESULT_SCHEDULE = {wid: {day_index: None for day_index in range(NUM_DAYS)} for wid in WORKER_IDS}
SOLVER_STATUS = solver.StatusName(status)

if status == cp_model.OPTIMAL or status == cp_model.FEASIBLE:
    for w in WORKER_IDS:
        for d in range(NUM_DAYS):
            for s in SHIFT_CODES:
                if solver.Value(x[(w, d, s)]) == 1:
                    RESULT_SCHEDULE[w][d] = s
